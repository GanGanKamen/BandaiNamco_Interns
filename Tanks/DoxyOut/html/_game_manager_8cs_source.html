132  // This is called from start and will run each phase of the game one after another.
133  private IEnumerator GameLoop()
134  {
135  // Start off by running the 'RoundStarting' coroutine but don't return until it's finished.
136  yield return StartCoroutine(RoundStarting());
137 
138  // Once the 'RoundStarting' coroutine is finished, run the 'RoundPlaying' coroutine but don't return until it's finished.
139  yield return StartCoroutine(RoundPlaying());
140 
141  // Once execution has returned here, run the 'RoundEnding' coroutine, again don't return until it's finished.
142  yield return StartCoroutine(RoundEnding());
143 
144  // This code is not run until 'RoundEnding' has finished. At which point, check if a game winner has been found.
145  if (GameWinner != null)
146  {
147  yield return StartCoroutine(GameResult());
148 
149  // If there is a game winner, restart the level.
150  SceneManager.LoadScene(0);
151  }
152  else
153  {
154  // If there isn't a winner yet, restart this coroutine so the loop continues.
155  // Note that this coroutine doesn't yield. This means that the current version of the GameLoop will end.
156  StartCoroutine(GameLoop());
157  }
158  }

254  // This is used to check if there is one or fewer tanks remaining and thus the round should end.
255  private bool OneTankLeft()
256  {
257  // Start the count of tanks left at zero.
258  int numTanksLeft = 0;
259 
260  // Go through all the tanks...
261  for (int i = 0; i < Tanks.Length; i++)
262  {
263  // ... and if they are active, increment the counter.
264  if (Tanks[i].Instance.activeSelf)
265  numTanksLeft++;
266  }
267 
268  // If there are one or fewer tanks remaining return true, otherwise return false.
269  return numTanksLeft <= 1;
270  }

275  private TankManager GetRoundWinner()

278  for (int i = 0; i < Tanks.Length; i++)
