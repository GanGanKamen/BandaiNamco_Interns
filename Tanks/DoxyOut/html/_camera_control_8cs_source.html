43  FindAveragePosition();
44

50  private void FindAveragePosition()
51  {
52  Vector3 averagePos = new Vector3 ();
53  int numTargets = 0;
54 
55  // Go through all the targets and add their positions together.
56  for (int i = 0; i < m_Targets.Length; i++)
57  {
58  // If the target isn't active, go on to the next one.
59  if (!m_Targets[i].gameObject.activeSelf)
60  continue;
61 
62  // Add to the average and increment the number of targets in the average.
63  averagePos += m_Targets[i].position;
64  numTargets++;
65  }
66 
67  // If there are targets divide the sum of the positions by the number of them to find the average.
68  if (numTargets > 0)
69  averagePos /= numTargets;
70 
71  // Keep the same y value.
72  averagePos.y = transform.position.y;
73 
74  // The desired position is the average position;
75  m_DesiredPosition = averagePos;
76  }

87  private float FindRequiredSize()
88  {
89  // Find the position the camera rig is moving towards in its local space.
90  Vector3 desiredLocalPos = transform.InverseTransformPoint(m_DesiredPosition);
91 
92  // Start the camera's size calculation at zero.
93  float size = 0f;
94 
95  // Go through all the targets...
96  for (int i = 0; i < m_Targets.Length; i++)
97  {
98  // ... and if they aren't active continue on to the next target.
99  if (!m_Targets[i].gameObject.activeSelf)
100  continue;
101 
102  // Otherwise, find the position of the target in the camera's local space.
103  Vector3 targetLocalPos = transform.InverseTransformPoint(m_Targets[i].position);
104 
105  // Find the position of the target from the desired position of the camera's local space.
106  Vector3 desiredPosToTarget = targetLocalPos - desiredLocalPos;
107 
108  // Choose the largest out of the current size and the distance of the tank 'up' or 'down' from the camera.
109  size = Mathf.Max(size, Mathf.Abs(desiredPosToTarget.y));
110 
111  // Choose the largest out of the current size and the calculated size based on the tank being to the left or right of the camera.
112  size = Mathf.Max(size, Mathf.Abs(desiredPosToTarget.x) / m_Camera.aspect);
113  }
114 
115  // Add the edge buffer to the size.
116  size += m_ScreenEdgeBuffer;
117 
118  // Make sure the camera's size isn't below the minimum.
119  size = Mathf.Max(size, m_MinSize);
120 
121  return size;
122  }
